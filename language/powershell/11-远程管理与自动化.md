function Get-ServerHealth {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)]
        [string[]]$ComputerName,
        
        [string]$ReportPath = ".\ServerHealth.html"
    )
    
    $results = Invoke-Command -ComputerName $ComputerName -ScriptBlock {
        # ç³»ç»Ÿä¿¡æ¯
        $os = Get-CimInstance Win32_OperatingSystem
        $cpu = Get-CimInstance Win32_Processor
        $disk = Get-CimInstance Win32_LogicalDisk -Filter "DriveType=3"
        
        # CPUä½¿ç”¨ç‡
        $cpuUsage = (Get-Counter '\Processor(_Total)\% Processor Time').CounterSamples.CookedValue
        
        # å†…å­˜ä½¿ç”¨
        $totalMemory = $os.TotalVisibleMemorySize
        $freeMemory = $os.FreePhysicalMemory
        $memoryUsage = [Math]::Round((($totalMemory - $freeMemory) / $totalMemory) * 100, 2)
        
        # æœåŠ¡çŠ¶æ€
        $services = Get-Service | Where-Object {$_.StartType -eq "Automatic"}
        $stoppedServices = $services | Where-Object {$_.Status -ne "Running"}
        
        [PSCustomObject]@{

            OS = $os.Caption
            LastBoot = $os.LastBootUpTime
            Uptime = (Get-Date) - $os.LastBootUpTime
            CPUUsage = [Math]::Round($cpuUsage, 2)
            MemoryUsagePercent = $memoryUsage
            DiskSpace = $disk | ForEach-Object {
                [PSCustomObject]@{
                    Drive = $_.DeviceID
                    FreeGB = [Math]::Round($_.FreeSpace / 1GB, 2)
                    TotalGB = [Math]::Round($_.Size / 1GB, 2)
                    PercentFree = [Math]::Round(($_.FreeSpace / $_.Size) * 100, 2)
                }
            }
            StoppedServices = $stoppedServices.Count
            StoppedServicesList = $stoppedServices.Name -join ", "
            Status = if ($cpuUsage -gt 90 -or $memoryUsage -gt 90) {"Critical"}
                    elseif ($cpuUsage -gt 75 -or $memoryUsage -gt 75) {"Warning"}
                    else {"OK"}
        }
    }
    
    # ç”ŸæˆHTMLæŠ¥å‘Š
    $html = $results | ConvertTo-Html -Head @"

    .warning {background-color: #ff9800; color: white;}
    .ok {background-color: #4CAF50; color: white;}
</style>
"@
    
    $html | Out-File $ReportPath

function Start-AutomatedBackup {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)]
        [string]$SourcePath,
        
        [Parameter(Mandatory=$true)]
        [string]$DestinationPath,
        
        [int]$RetentionDays = 7
    )
    
    $timestamp = Get-Date -Format "yyyyMMdd_HHmmss"
    $backupFolder = Join-Path $DestinationPath "Backup_$timestamp"
    
    Write-Host "=== Starting Backup ===" -ForegroundColor Cyan
    Write-Host "Source: $SourcePath"
    Write-Host "Destination: $backupFolder"
    
    try {
        # åˆ›å»ºå¤‡ä»½ç›®å½•
        New-Item -ItemType Directory -Path $backupFolder -Force | Out-Null
        
        # å¤åˆ¶æ–‡ä»¶
        $files = Get-ChildItem -Path $SourcePath -Recurse -File
        $total = $files.Count
        $current = 0
        
        foreach ($file in $files) {
            $current++
            Write-Progress -Activity "Backing up files" `
                -Status "$current of $total" `
                -PercentComplete ($current/$total*100)
            
            $relativePath = $file.FullName.Substring($SourcePath.Length)
            $targetPath = Join-Path $backupFolder $relativePath
            $targetDir = Split-Path $targetPath -Parent
            
            if (-not (Test-Path $targetDir)) {
                New-Item -ItemType Directory -Path $targetDir -Force | Out-Null
            }
            
            Copy-Item -Path $file.FullName -Destination $targetPath
        }
        
        Write-Progress -Activity "Backing up files" -Completed
        
        # æ¸…ç†æ—§å¤‡ä»½
        $cutoffDate = (Get-Date).AddDays(-$RetentionDays)
        Get-ChildItem -Path $DestinationPath -Directory | 
            Where-Object {$_.Name -match "^Backup_" -and $_.CreationTime -lt $cutoffDate} |
            ForEach-Object {
                Write-Host "Removing old backup: $($_.Name)" -ForegroundColor Yellow
                Remove-Item $_.FullName -Recurse -Force
            }
        
        Write-Host "âœ“ Backup completed successfully" -ForegroundColor Green
        Write-Host "Files backed up: $total"
        
        # å‘é€é€šçŸ¥é‚®ä»¶
        # Send-MailMessage ...
    }
    catch {
        Write-Host "âœ— Backup failed: $_" -ForegroundColor Red
        throw
    }
}

function Invoke-WindowsUpdate {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)]
        [string[]]$ComputerName,
        
        [switch]$AutoReboot
    )
    
    $results = Invoke-Command -ComputerName $ComputerName -ScriptBlock {

            Computer = $env:COMPUTERNAME
            UpdatesFound = $searchResult.Updates.Count

            Status = "Updates installed"
        }
