import requests
from bs4 import BeautifulSoup
import pandas as pd
from datetime import datetime
import time

class NewsScraper:
    def __init__(self):
        self.headers = {
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        }
        self.news_list = []
    
    def fetch_page(self, url):
        """è·å–ç½‘é¡µå†…å®¹"""
        try:
            response = requests.get(url, headers=self.headers, timeout=10)
            response.encoding = response.apparent_encoding
            return response.text
        except Exception as e:
            print(f"é”™è¯¯: {e}")
            return None
    
    def parse_news(self, html, source):
        """è§£ææ–°é—»(éœ€æ ¹æ®å®é™…ç½‘ç«™ç»“æ„è°ƒæ•´)"""
        soup = BeautifulSoup(html, 'html.parser')
        
        # ç¤ºä¾‹è§£æé€»è¾‘(éœ€è¦æ ¹æ®ç›®æ ‡ç½‘ç«™ä¿®æ”¹)
        articles = soup.find_all('article', class_='news-item')
        
        for article in articles:
            try:
                title = article.find('h2').get_text(strip=True)
                link = article.find('a')['href']
                summary = article.find('p', class_='summary').get_text(strip=True)
                
                news_item = {
                    'title': title,
                    'link': link,
                    'summary': summary,
                    'source': source,
                    'scraped_at': datetime.now().isoformat()
                }
                
                self.news_list.append(news_item)
            except Exception as e:
                continue
    
    def scrape_multiple_sources(self, sources):
        """çˆ¬å–å¤šä¸ªæ–°é—»æº"""
        for source_name, source_url in sources.items():
            print(f"æ­£åœ¨çˆ¬å–: {source_name}")
            html = self.fetch_page(source_url)
            if html:
                self.parse_news(html, source_name)
            time.sleep(2)  # å»¶è¿Ÿé¿å…è¢«å°
    
    def remove_duplicates(self):
        """å»é‡"""
        seen = set()
        unique_news = []
        
        for news in self.news_list:
            if news['title'] not in seen:
                seen.add(news['title'])
                unique_news.append(news)
        
        self.news_list = unique_news
        print(f"å»é‡åå‰©ä½™ {len(self.news_list)} æ¡æ–°é—»")
    
    def save_to_csv(self, filename='news.csv'):
        """ä¿å­˜ä¸ºCSV"""
        df = pd.DataFrame(self.news_list)
        df.to_csv(filename, index=False, encoding='utf-8-sig')
        print(f"âœ“ å·²ä¿å­˜åˆ° {filename}")
    
    def save_to_json(self, filename='news.json'):
        """ä¿å­˜ä¸ºJSON"""
        import json
        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(self.news_list, f, ensure_ascii=False, indent=2)
        print(f"âœ“ å·²ä¿å­˜åˆ° {filename}")
    
    def generate_report(self):
        """ç”Ÿæˆç»Ÿè®¡æŠ¥å‘Š"""
        if not self.news_list:
            print("æ²¡æœ‰æ–°é—»æ•°æ®")
            return
        
        df = pd.DataFrame(self.news_list)
        
        print("\n" + "=" * 50)
        print("æ–°é—»çˆ¬å–ç»Ÿè®¡æŠ¥å‘Š")
        print("=" * 50)
        print(f"æ€»æ–°é—»æ•°: {len(df)}")
        print(f"\nå„æ¥æºæ–°é—»æ•°:")
        print(df['source'].value_counts())

import pandas as pd

    def __init__(self, data_file='finance.csv'):
        self.data_file = data_file
        self.load_data()
    
    def load_data(self):
        """åŠ è½½æ•°æ®"""
        try:
            self.df = pd.read_csv(self.data_file, parse_dates=['date'])

    def save_data(self):

        new_record = {

            'type': trans_type,  # 'income' or 'expense'
            'category': category,
            'amount': amount,
            'description': description
        }
        
        self.df = pd.concat([self.df, pd.DataFrame([new_record])],

            (self.df['date'].dt.year == year) & 
            (self.df['date'].dt.month == month)

        total_income = monthly[monthly['type'] == 'income']['amount'].sum()
        total_expense = monthly[monthly['type'] == 'expense']['amount'].sum()

        print(f"{year}å¹´{month}æœˆè´¢åŠ¡æŠ¥è¡¨")
        print(f"{'='*50}")
        print(f"æ€»æ”¶å…¥: Â¥{total_income:,.2f}")

        print(f"ç»“ä½™:   Â¥{balance:,.2f}")
        print(f"{'='*50}\n")

    def category_analysis(self, trans_type='expense'):
        """åˆ†ç±»åˆ†æ"""
        filtered = self.df[self.df['type'] == trans_type]

from flask import Flask, jsonify, request
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

app = Flask(__name__)

app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
db = SQLAlchemy(app)

# æ¨¡å‹å®šä¹‰
class Book(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(200), nullable=False)
    author = db.Column(db.String(100), nullable=False)
    isbn = db.Column(db.String(13), unique=True, nullable=False)
    published_date = db.Column(db.Date)
    price = db.Column(db.Float)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)
    
    def to_dict(self):
        return {
            'id': self.id,
            'title': self.title,
            'author': self.author,
            'isbn': self.isbn,
            'published_date': self.published_date.isoformat() if self.published_date else None,
            'price': self.price
        }

# åˆ›å»ºæ•°æ®åº“
with app.app_context():
    db.create_all()

# APIè·¯ç”±

@app.route('/api/books', methods=['GET'])
def get_books():
    """è·å–æ‰€æœ‰å›¾ä¹¦"""
    books = Book.query.all()
    return jsonify([book.to_dict() for book in books])

@app.route('/api/books/<int:book_id>', methods=['GET'])
def get_book(book_id):
    """è·å–å•æœ¬å›¾ä¹¦"""
    book = Book.query.get_or_404(book_id)
    return jsonify(book.to_dict())

@app.route('/api/books', methods=['POST'])
def create_book():
    """åˆ›å»ºå›¾ä¹¦"""
    data = request.get_json()
    
    book = Book(
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        published_date=datetime.strptime(data.get('published_date'), '%Y-%m-%d').date() if data.get('published_date') else None,
        price=data.get('price')
    )
    
    db.session.add(book)
    db.session.commit()
    
    return jsonify(book.to_dict()), 201

@app.route('/api/books/<int:book_id>', methods=['PUT'])
def update_book(book_id):
    """æ›´æ–°å›¾ä¹¦"""
    book = Book.query.get_or_404(book_id)
    data = request.get_json()
    
    book.title = data.get('title', book.title)
    book.author = data.get('author', book.author)
    book.isbn = data.get('isbn', book.isbn)
    book.price = data.get('price', book.price)
    
    db.session.commit()
    
    return jsonify(book.to_dict())

@app.route('/api/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id):
    """åˆ é™¤å›¾ä¹¦"""
    book = Book.query.get_or_404(book_id)
    db.session.delete(book)
    db.session.commit()
    
    return jsonify({'message': 'Book deleted successfully'}), 200

@app.route('/api/books/search', methods=['GET'])
def search_books():
    """æœç´¢å›¾ä¹¦"""
    keyword = request.args.get('q', '')
    books = Book.query.filter(
        (Book.title.contains(keyword)) | 
        (Book.author.contains(keyword))
    ).all()
    
    return jsonify([book.to_dict() for book in books])

@app.errorhandler(404)
def not_found(error):
    return jsonify({'error': 'Not found'}), 404

@app.errorhandler(500)
def internal_error(error):
    return jsonify({'error': 'Internal server error'}), 500

if __name__ == '__main__':
    app.run(debug=True)
