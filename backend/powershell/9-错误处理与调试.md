function Read-FileRobust {
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)]
        [string]$Path,
        
        [int]$RetryCount = 3,
        [int]$RetryDelay = 1000
    )
    
    Write-Verbose "Attempting to read file: $Path"
    
    # å‚æ•°éªŒè¯
    if ([string]::IsNullOrWhiteSpace($Path)) {
        throw [ArgumentException] "Path cannot be empty"
    }
    
    $attempt = 0
    $success = $false
    $content = $null
    
    while ((-not $success) -and ($attempt -lt $RetryCount)) {
        $attempt++
        
        try {
            Write-Verbose "Attempt $attempt of $RetryCount"
            
            # æ£€æŸ¥æ–‡ä»¶æ˜¯å¦å­˜åœ¨
            if (-not (Test-Path $Path)) {
                throw [System.IO.FileNotFoundException] "File not found: $Path"
            }
            
            # æ£€æŸ¥æ˜¯å¦æœ‰è¯»å–æƒé™
            $file = Get-Item $Path
            if ($file.Attributes -match "Encrypted") {
                throw [UnauthorizedAccessException] "File is encrypted"
            }
            
            # è¯»å–æ–‡ä»¶
            $content = Get-Content -Path $Path -ErrorAction Stop
            $success = $true
            
            Write-Verbose "Successfully read file"
        }
        catch [System.IO.FileNotFoundException] {
            Write-Error "File not found: $Path"
            break  # ä¸é‡è¯•
        }
        catch [System.UnauthorizedAccessException] {
            Write-Error "Access denied: $Path"
            break  # ä¸é‡è¯•
        }
        catch [System.IO.IOException] {
            Write-Warning "IO error on attempt $attempt`: $_"
            if ($attempt -lt $RetryCount) {
                Write-Verbose "Retrying in $($RetryDelay)ms..."
                Start-Sleep -Milliseconds $RetryDelay
            }
        }
        catch {
            Write-Error "Unexpected error: $_"
            break  # ä¸é‡è¯•æœªçŸ¥é”™è¯¯
        }
    }
    
    if (-not $success) {
        throw "Failed to read file after $RetryCount attempts"
    }
    
    return $content
}
