function Get-FileHashReport {
    <#
    .SYNOPSIS
        ç”Ÿæˆç›®å½•æ–‡ä»¶å“ˆå¸ŒæŠ¥å‘Š
    .PARAMETER Path
        ç›®å½•è·¯å¾„
    .PARAMETER Algorithm
        å“ˆå¸Œç®—æ³•
    #>
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)]
        [string]$Path,
        
        [ValidateSet("MD5","SHA1","SHA256","SHA512")]
        [string]$Algorithm = "SHA256"
    )
    
    Get-ChildItem -Path $Path -Recurse -File | ForEach-Object {
        $hash = Get-FileHash -Path $_.FullName -Algorithm $Algorithm
        
        [PSCustomObject]@{
            File = $_.Name
            Path = $_.FullName
            Algorithm = $Algorithm
            Hash = $hash.Hash
            Size = $_.Length
            Modified = $_.LastWriteTime
        }
    }
}

# Public/Compare-Directory.ps1
function Compare-Directory {
    <#
    .SYNOPSIS
        æ¯”è¾ƒä¸¤ä¸ªç›®å½•
    #>
    [CmdletBinding()]
    param(
        [Parameter(Mandatory=$true)]
        [string]$Path1,
        
        [Parameter(Mandatory=$true)]
        [string]$Path2
    )
    
    $files1 = Get-ChildItem -Path $Path1 -Recurse -File
    $files2 = Get-ChildItem -Path $Path2 -Recurse -File
    
    $result = @{
        OnlyInPath1 = @()
        OnlyInPath2 = @()
        Different = @()
        Identical = @()
    }
    
    foreach ($file1 in $files1) {
        $relativePath = $file1.FullName.Substring($Path1.Length)
        $file2Path = Join-Path $Path2 $relativePath
        
        if (Test-Path $file2Path) {
            $file2 = Get-Item $file2Path
            $hash1 = Get-FileHash $file1.FullName
            $hash2 = Get-FileHash $file2.FullName
            
            if ($hash1.Hash -eq $hash2.Hash) {
                $result.Identical += $relativePath
            } else {
                $result.Different += $relativePath
            }
        } else {
            $result.OnlyInPath1 += $relativePath
        }
    }
    
    foreach ($file2 in $files2) {
        $relativePath = $file2.FullName.Substring($Path2.Length)
        $file1Path = Join-Path $Path1 $relativePath
        
        if (-not (Test-Path $file1Path)) {
            $result.OnlyInPath2 += $relativePath
        }
    }
    
    return $result
}
