function Invoke-Calculator {
    param(
        [Parameter(Mandatory)]
        [double]$Num1,
        
        [Parameter(Mandatory)]
        [ValidateSet('+','-','*','/','%','^')]
        [string]$Operator,
        
        [Parameter(Mandatory)]
        [double]$Num2
    )
    
    $result = switch ($Operator) {
        '+' { $Num1 + $Num2 }
        '-' { $Num1 - $Num2 }
        '*' { $Num1 * $Num2 }
        '/' { 
            if ($Num2 -eq 0) {
                Write-Error "Division by zero!"
                return
            }
            $Num1 / $Num2 
        }
        '%' { $Num1 % $Num2 }
        '^' { [Math]::Pow($Num1, $Num2) }
    }
    
    Write-Host "$Num1 $Operator $Num2 = $result" -ForegroundColor Green
    return $result
}
