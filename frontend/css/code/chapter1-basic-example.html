/* 颜色表示法示例 */ .color-red { background-color: red; /* 颜色关键字 */ } .color-hex { background-color: #00FF00; /* 十六进制 */ } .color-rgb { background-color: rgb(0, 0, 255); /* RGB */ } .color-rgba { background-color: rgba(255, 0, 0, 0.7); /* RGBA带透明度 */ } .color-hsl { background-color: hsl(60, 100%, 50%); /* HSL */ } .color-hsla { background-color: hsla(180, 100%, 50%, 0.8); /* HSLA带透明度 */ }

/* 字体与文本样式示例 */ .font-family { font-family: 'Microsoft YaHei', Arial, sans-serif; } .font-size { font-size: 18px; /* 也可以使用rem, em, % */ } .font-weight { font-weight: bold; /* 也可以使用数值100-900 */ } .font-style { font-style: italic; } .text-align { text-align: center; /* left, right, justify */ } .text-decoration { text-decoration: underline; /* none, line-through, overline */ } .text-transform { text-transform: uppercase; /* lowercase, capitalize */ } .line-height { line-height: 1.6; /* 推荐使用无单位值 */ }

3. CSS盒模型

标准盒模型 vs IE盒模型

标准盒模型

width: 200px + padding: 40px + border: 10px

总宽度: 250px

IE盒模型

width: 200px (包含padding和border)

内容宽度: 150px

/* 盒模型示例 */ /* 标准盒模型（默认） */ .content-box { box-sizing: content-box; width: 200px; height: 150px; padding: 20px; border: 5px solid #e74c3c; /* 实际宽度 = 200 + 20*2 + 5*2 = 250px */ } /* IE盒模型（推荐） */ .border-box { box-sizing: border-box; width: 200px; height: 150px; padding: 20px; border: 5px solid #27ae60; /* 内容宽度 = 200 - 20*2 - 5*2 = 150px */ } /* 为所有元素设置IE盒模型（最佳实践） */ * { box-sizing: border-box; }

/* 响应式卡片组件 */ .card { background-color: white; border-radius: 10px; overflow: hidden; box-shadow: 0 4px 15px rgba(0, 0, 0, 0.1); transition: transform 0.3s ease, box-shadow 0.3s ease; } /* 卡片悬停效果 */ .card:hover { transform: translateY(-10px); box-shadow: 0 10px 25px rgba(0, 0, 0, 0.15); } /* 卡片图片 */ .card-image { height: 200px; overflow: hidden; } .card-image img { width: 100%; height: 100%; object-fit: cover; transition: transform 0.5s ease; } .card:hover .card-image img { transform: scale(1.1); } /* 卡片内容 */ .card-content { padding: 20px; } /* 响应式布局 */ .card-container { display: grid; grid-template-columns: repeat(auto-fill, minmax(300px, 1fr)); gap: 30px; } /* 媒体查询 */ @media (max-width: 768px) { .card-container { grid-template-columns: repeat(auto-fill, minmax(250px, 1fr)); gap: 20px; } }
