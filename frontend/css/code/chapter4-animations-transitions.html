多属性过渡
/* 基础过渡效果示例 */ /* 单属性过渡 */ .basic-transition { transition: background-color 0.3s ease; } .basic-transition:hover { background-color: #e74c3c; } /* 多属性过渡 */ .multi-transition { transition: background-color 0.5s ease, transform 0.3s ease, border-radius 0.5s ease; } .multi-transition:hover { background-color: #2ecc71; transform: scale(1.1) rotate(5deg); border-radius: 50%; } /* 简写形式 */ .element { /* property duration timing-function delay */ transition: all 0.3s ease 0s; } /* 分别定义多个过渡 */ .element { transition-property: background-color, transform; transition-duration: 0.5s, 0.3s; transition-timing-function: ease, linear; transition-delay: 0s, 0.2s; }

/* 过渡时间函数示例 */ /* 预定义时间函数 */ .ease-transition { transition: all 1s ease; /* 慢-快-慢 */ } .linear-transition { transition: all 1s linear; /* 匀速 */ } .ease-in-transition { transition: all 1s ease-in; /* 慢开始 */ } .ease-out-transition { transition: all 1s ease-out; /* 慢结束 */ } .ease-in-out-transition { transition: all 1s ease-in-out; /* 慢开始和结束 */ } /* 自定义贝塞尔曲线 */ .custom-cubic-transition { transition: all 1s cubic-bezier(0.68, -0.55, 0.265, 1.55); /* 可以创建回弹效果 */ } /* 常见的自定义贝塞尔曲线 */ .ease-out-quad { transition: all 0.3s cubic-bezier(0.25, 0.46, 0.45, 0.94); } .ease-out-back { transition: all 0.3s cubic-bezier(0.175, 0.885, 0.32, 1.275); }

/* 2D变换效果示例 */ /* 平移变换 */ .translate-transform:hover { transform: translate(20px, -10px); /* X轴, Y轴 */ /* 也可以单独设置：translateX(), translateY() */ } /* 缩放变换 */ .scale-transform:hover { transform: scale(1.2); /* 同时缩放X和Y */ /* 也可以单独设置：scaleX(), scaleY() */ } /* 旋转变换 */ .rotate-transform:hover { transform: rotate(45deg); /* 正值顺时针，负值逆时针 */ } /* 倾斜变换 */ .skew-transform:hover { transform: skew(10deg, 5deg); /* X轴倾斜，Y轴倾斜 */ /* 也可以单独设置：skewX(), skewY() */ } /* 组合变换 */ .combo-transform:hover { transform: translate(10px, -10px) rotate(15deg) scale(1.1); /* 注意变换的顺序会影响最终结果 */ } /* 设置变换原点 */ .transform-origin-center { transform-origin: center; } .transform-origin-top-left { transform-origin: top left; } .transform-origin-bottom-right { transform-origin: bottom right; }

/* 基础关键帧动画示例 */ /* 定义关键帧 */ @keyframes fadeIn { from { opacity: 0; transform: translateY(-20px); } to { opacity: 1; transform: translateY(0); } } /* 使用动画 */ .fade-in-animation { animation: fadeIn 1s ease-out; } /* 多步骤关键帧 */ @keyframes slideIn { 0% { transform: translateX(-100%); opacity: 0; } 50% { transform: translateX(50px); opacity: 0.8; } 100% { transform: translateX(0); opacity: 1; } } /* 无限循环动画 */ @keyframes pulse { 0% { transform: scale(1); background-color: #3498db; } 50% { transform: scale(1.1); background-color: #2ecc71; } 100% { transform: scale(1); background-color: #3498db; } } .pulse-animation { animation: pulse 2s ease-in-out infinite; } /* 动画属性完整定义 */ .animation-example { animation-name: fadeIn; animation-duration: 1s; animation-timing-function: ease-out; animation-delay: 0s; animation-iteration-count: 1; animation-direction: normal; animation-fill-mode: none; animation-play-state: running; } /* 简写形式 */ .animation-shorthand { /* name duration timing-function delay iteration-count direction fill-mode */ animation: fadeIn 1s ease-out 0s 1 normal none; }

/* 动画控制与高级技巧 */ /* 暂停/播放动画 */ .pause-resume-animation { animation: rotate 3s linear infinite; } .pause-resume-animation:hover { animation-play-state: paused; /* 可以通过JavaScript控制：element.style.animationPlayState = 'paused' */ } /* 延迟动画 */ .bounce-animation { animation: bounce 2s ease-in-out infinite 1s; /* 最后一个值是延迟时间 */ } /* 交替动画 */ .alternate-animation { animation: slideHorizontal 2s ease-in-out infinite alternate; /* alternate: 正向-反向-正向... */ /* alternate-reverse: 反向-正向-反向... */ } /* 填充模式 */ .fill-mode-animation { animation: fadeOut 2s ease-in-out forwards; /* none: 默认值，动画结束后回到初始状态 */ /* forwards: 保持动画结束状态 */ /* backwards: 应用第一个关键帧的样式到动画开始前 */ /* both: 同时应用forwards和backwards */ } /* 多重动画 */ .multiple-animations { animation: slideHorizontal 3s ease-in-out infinite alternate, colorChange 6s ease-in-out infinite; /* 可以同时应用多个动画 */ } /* 性能优化：使用GPU加速 */ .gpu-accelerated { transform: translateZ(0); backface-visibility: hidden; perspective: 1000px; } /* 动画性能最佳实践 */ .optimized-animation { animation: optimizedMove 1s ease-out; } @keyframes optimizedMove { /* 优先使用transform和opacity，避免重排 */ from { transform: translateY(0); opacity: 1; } to { transform: translateY(100px); opacity: 0; } }
